package lesson5;

public class Employees {
    private String FIO;
    private int age;
    private String email;

    


    public int getAge() {  
        return age;
    }

    public Employees(String FIO, String email, int age) {
        this.FIO = FIO;
        this.email = email;
        this.age = age;
    }

    public void printInfo(){
        System.out.println("FIO "+FIO+"email"+email+"age "+age);
    }
}
